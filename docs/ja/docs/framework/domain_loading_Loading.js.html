                                                                                                                                                                                            /**
 * @class
 */
export class Loading {
    /**
     * @description デフォルトのローディング演出を開始
     *              Starts default loading direction
     *
     * @return {void}
     * @method
     * @public
     */
    start() {
        // @ts-ignore
        const config = next2d.fw.config;
        if (!config.loading) {
            return;
        }
        const callback = config.loading.callback;
        if (!callback) {
            return;
        }
        // @ts-ignore
        const packages = next2d.fw.packages;
        // @ts-ignore
        const parser = next2d.fw.parser;
        const name = parser.execute(callback);
        const CallbackClass = packages.has(name)
            ? packages.get(name)
            // @ts-ignore
            : next2d.fw.DefaultLoading;
        new CallbackClass().start();
    }
    /**
     * @description デフォルトのローディング演出を終了
     *              End default loading direction
     *
     * @return {void}
     * @method
     * @public
     */
    end() {
        // @ts-ignore
        const config = next2d.fw.config;
        if (!config.loading) {
            return;
        }
        const callback = config.loading.callback;
        if (!callback) {
            return;
        }
        // @ts-ignore
        const packages = next2d.fw.packages;
        // @ts-ignore
        const parser = next2d.fw.parser;
        const name = parser.execute(callback);
        const CallbackClass = packages.has(name)
            ? packages.get(name)
            // @ts-ignore
            : next2d.fw.DefaultLoading;
        new CallbackClass().end();
    }
}
