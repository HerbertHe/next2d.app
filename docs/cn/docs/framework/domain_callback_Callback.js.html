                                                                                                                                                                                                                                        /**
 * @class
 * @memberof domain.callback
 */
export class Callback {
    /**
     * @description configで指定されたクラスのexecute関数を実行
     *              Execute function of the class specified in config.
     *
     * @param  {string|array} [callback=""]
     * @param  {*} [value=null]
     * @return {Promise}
     * @method
     * @public
     */
    execute(callback = "", value = null) {
        if (!callback) {
            return Promise.resolve();
        }
        const callbacks = typeof callback === "string"
            ? [callback]
            : callback;
        const promises = [];
        // @ts-ignore
        const packages = next2d.fw.packages;
        // @ts-ignore
        const parser = next2d.fw.parser;
        for (let idx = 0; idx < callbacks.length; ++idx) {
            const name = parser.execute(callbacks[idx]);
            if (!packages.has(name)) {
                continue;
            }
            const CallbackClass = packages.get(name);
            promises.push(new CallbackClass().execute(value));
        }
        return Promise.all(promises);
    }
}
