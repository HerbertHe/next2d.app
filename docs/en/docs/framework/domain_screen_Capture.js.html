                                                                                                                                                                                                                                        /**
 * @class
 * @memberof domain.screen
 */
export class Capture {
    /**
     * @description 現時点の描画キャプチャーを生成
     *              Generate current drawing capture
     *
     * @return {Promise<void>}
     * @method
     * @public
     */
    execute() {
        return new Promise((resolve) => {
            // @ts-ignore
            const context = next2d.fw.context;
            // @ts-ignore
            const root = context.root;
            // @ts-ignore
            const { Shape } = next2d.display;
            // @ts-ignore
            const config = next2d.fw.config;
            const width = config.stage.width;
            const height = config.stage.height;
            const mask = new Shape();
            mask
                .graphics
                .beginFill(0, 0.8)
                .drawRect(0, 0, width, height)
                .endFill();
            const player = root.stage._$player;
            const matrix = player._$matrix;
            const tx = matrix[4];
            if (tx) {
                const scaleX = matrix[0];
                mask.scaleX = (width + tx * 2 / scaleX) / width;
                mask.x = -tx / scaleX;
            }
            const ty = matrix[5];
            if (ty) {
                const scaleY = matrix[3];
                mask.scaleY = (height + ty * 2 / scaleY) / height;
                mask.y = -ty / scaleY;
            }
            root.mouseChildren = false;
            root.addChild(mask);
            setTimeout(() => {
                resolve();
            }, 350);
        });
    }
}
